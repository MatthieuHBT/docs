---
title: Lister les webhooks
api: GET https://roapix.up.railway.app/api/v1/webhooks
description: Retourne tous les webhooks enregistrés pour votre workspace.
---

<ParamField header="Authorization" type="string" required>
  `Bearer roapix_votre_clé`
</ParamField>

<ResponseField name="data" type="array">
  <Expandable title="Propriétés de chaque item">
    <ResponseField name="id" type="string">UUID du webhook</ResponseField>
    <ResponseField name="url" type="string">URL de destination</ResponseField>
    <ResponseField name="events" type="array[string]">Événements écoutés</ResponseField>
    <ResponseField name="is_active" type="boolean">Actif ou non</ResponseField>
    <ResponseField name="created_at" type="string">Date de création</ResponseField>
    <ResponseField name="last_triggered_at" type="string | null">Dernière fois déclenché</ResponseField>
  </Expandable>
</ResponseField>

<RequestExample>
```bash cURL
curl https://roapix.up.railway.app/api/v1/webhooks \
  -H "Authorization: Bearer roapix_votre_clé"
```
</RequestExample>
